## 四、一键部署脚本

将以下内容保存为 `deploy_models.sh`，在 adb 连接机顶盒后执行：

```bash
#!/bin/bash
# ============================================================
# JOCTV IPTV Agent — 机顶盒模型一键部署脚本
# 用法: ./deploy_models.sh [本地模型根目录]
# 默认模型根目录: 当前项目根目录
# ============================================================

set -euo pipefail

MODEL_ROOT="${1:-.}"

echo "=== JOCTV IPTV Agent 模型部署 ==="
echo "模型根目录: $MODEL_ROOT"
echo ""

# 检查 adb 连接
if ! adb get-state 2>/dev/null | grep -q "device"; then
    echo "错误: adb 未连接到设备，请先连接机顶盒"
    exit 1
fi

echo "--- [1/4] 推送 ASR 模型 ---"
ASR_DIR="sherpa-onnx-rk3576-20-seconds-sense-voice-zh-en-ja-ko-yue-2024-07-17"
if [ -d "$MODEL_ROOT/$ASR_DIR" ]; then
    adb push "$MODEL_ROOT/$ASR_DIR/" "/sdcard/$ASR_DIR/"
    echo "ASR 推送完成"
else
    echo "跳过 ASR: 目录不存在 $MODEL_ROOT/$ASR_DIR"
fi

echo ""
echo "--- [2/4] 推送 TTS 模型 ---"
adb shell mkdir -p /sdcard/iptv-agent-models/tts

for tts_dir in bert-vits2-MNN vits-melo-tts-zh_en kokoro-en-v0_19; do
    if [ -d "$MODEL_ROOT/$tts_dir" ]; then
        adb shell mkdir -p "/sdcard/iptv-agent-models/tts/$tts_dir"
        adb push "$MODEL_ROOT/$tts_dir/" "/sdcard/iptv-agent-models/tts/$tts_dir/"
        echo "TTS [$tts_dir] 推送完成"
    else
        echo "跳过 TTS [$tts_dir]: 不存在"
    fi
done

echo ""
echo "--- [3/4] 推送数字人模型 ---"
# TaoAvatar 3D
if [ -d "$MODEL_ROOT/external/taoavatar-models" ]; then
    adb shell mkdir -p /sdcard/iptv-agent-models/taoavatar/TaoAvatar-NNR-MNN
    adb shell mkdir -p /sdcard/iptv-agent-models/taoavatar/UniTalker-MNN
    adb push "$MODEL_ROOT/external/taoavatar-models/TaoAvatar-NNR-MNN/" /sdcard/iptv-agent-models/taoavatar/TaoAvatar-NNR-MNN/
    adb push "$MODEL_ROOT/external/taoavatar-models/UniTalker-MNN/" /sdcard/iptv-agent-models/taoavatar/UniTalker-MNN/
    echo "TaoAvatar 3D 推送完成"
else
    echo "跳过 TaoAvatar: 目录不存在"
fi

# 2D 数字人
if [ -f "$MODEL_ROOT/avatar.glb" ]; then
    adb shell mkdir -p /sdcard/iptv-agent-models/digital-human
    adb push "$MODEL_ROOT/avatar.glb" /sdcard/iptv-agent-models/digital-human/avatar.glb
    echo "2D 数字人推送完成"
else
    echo "跳过 2D 数字人: avatar.glb 不存在"
fi

echo ""
echo "--- [4/4] 推送视频源 ---"
if [ -f "$MODEL_ROOT/1.mp4" ]; then
    adb push "$MODEL_ROOT/1.mp4" /sdcard/1.mp4
    echo "视频推送完成"
else
    echo "跳过视频: 1.mp4 不存在"
fi

echo ""
echo "=== 部署完成 ==="
echo "验证命令: adb shell ls -la /sdcard/sherpa-onnx-rk3576-*/ && adb shell ls -la /sdcard/iptv-agent-models/"
```
