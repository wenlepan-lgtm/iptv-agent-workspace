## 七、一键部署脚本（新机顶盒用）

```bash
#!/bin/bash
set -euo pipefail

ROOT="/Users/ala/工作项目/agent"

echo "=== JOCTV IPTV Agent 模型部署 ==="

if ! adb get-state 2>/dev/null | grep -q "device"; then
    echo "错误: adb 未连接，请先连接机顶盒"
    exit 1
fi

# 1. ASR
echo "--- [1/4] ASR SenseVoice ---"
adb push "$ROOT/sherpa-onnx-rk3576-20-seconds-sense-voice-zh-en-ja-ko-yue-2024-07-17/" \
  "/sdcard/sherpa-onnx-rk3576-20-seconds-sense-voice-zh-en-ja-ko-yue-2024-07-17/"

# 2. TTS — 中文 + 英文（整个 tts 目录一起推）
echo "--- [2/4] TTS 模型 ---"
adb shell mkdir -p /sdcard/iptv-agent-models/tts
# bert-vits2-MNN, matcha-icefall-zh-en, vits-zh-hf-fanchen-C, piper-en-us-lessac
# 按实际模型目录逐一推送
for dir in bert-vits2-MNN matcha-icefall-zh-en vits-zh-hf-fanchen-C piper-en-us-lessac; do
    if [ -d "$ROOT/$dir" ]; then
        adb shell mkdir -p "/sdcard/iptv-agent-models/tts/$dir"
        adb push "$ROOT/$dir/" "/sdcard/iptv-agent-models/tts/$dir/"
        echo "TTS [$dir] 推送完成"
    else
        echo "跳过 TTS [$dir]: 本地不存在"
    fi
done

# 3. 3D 数字人
echo "--- [3/4] TaoAvatar 3D ---"
adb shell mkdir -p /sdcard/iptv-agent-models/taoavatar/TaoAvatar-NNR-MNN
adb shell mkdir -p /sdcard/iptv-agent-models/taoavatar/UniTalker-MNN
adb push "$ROOT/iptv-edge-agent/external/taoavatar-models/TaoAvatar-NNR-MNN/" \
  /sdcard/iptv-agent-models/taoavatar/TaoAvatar-NNR-MNN/
adb push "$ROOT/iptv-edge-agent/external/taoavatar-models/UniTalker-MNN/" \
  /sdcard/iptv-agent-models/taoavatar/UniTalker-MNN/

# 4. 视频
echo "--- [4/4] 视频源 ---"
adb push "$ROOT/1.mp4" /sdcard/1.mp4

echo ""
echo "=== 部署完成 ==="
echo "验证: adb shell ls /sdcard/sherpa-onnx-rk3576-*/model.rknn"
echo "验证: adb shell ls /sdcard/iptv-agent-models/tts/matcha-icefall-zh-en/model-steps-3.onnx"
echo "验证: adb shell ls /sdcard/iptv-agent-models/taoavatar/UniTalker-MNN/audio2verts.mnn"
```
